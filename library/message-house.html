The Message House

Every message ladders up to Commercial Health — pick a pillar to see what to say

Library › The Message House

Interactive navigator for the Commercial Messaging Framework. Click a pillar to filter the toolkit to only the on-brand material for it.

Apex · 5 pillars · 8 reservoirs

The rule of the framework is simple: if a message doesn’t ladder up to Commercial Health, it’s off-brand. This tool makes that rule usable — choose a pillar (one of the five commercial questions) and it surfaces only the beliefs, problems, questions, myths, jobs-to-be-done, outcomes, and signature ideas that serve it.

How to use it. Pick the pillar you want to write about — the panel shows every belief, problem, question, myth, job-to-be-done, outcome, and signature idea that ladders to it. Everything you see is, by construction, on-brand. Turn any item into a post with the recipe in The Sigmoid Editorial System.

<script>
(function () {
  function esc(s){ return String(s).replace(/&/g,"&amp;").replace(/</g,"&lt;").replace(/>/g,"&gt;"); }
  function init() {
    var el = document.getElementById("mh-data"); if (!el) return;
    var D = JSON.parse(el.textContent);
    document.getElementById("mh-apex-line").textContent = D.apex.line;
    document.getElementById("mh-base").textContent = D.base;
    document.getElementById("mh-foundation").innerHTML =
      "<strong>Raw material:</strong> " + D.toolkit.map(function(c){ return c.label; }).join(" · ");

    var pillars = document.getElementById("mh-pillars");
    D.pillars.forEach(function (p) {
      var b = document.createElement("button");
      b.className = "mh-pillar"; b.setAttribute("data-pillar", p.id); b.setAttribute("aria-pressed", "false");
      b.innerHTML = '<div class="mh-p-name">' + esc(p.name) + '</div><div class="mh-p-sub">' + esc(p.sub) +
        '</div><div class="mh-p-q">' + esc(p.question) + '</div>';
      pillars.appendChild(b);
    });

    var apexBtn = document.getElementById("mh-apex");
    var detail = document.getElementById("mh-detail");

    function clearPressed() {
      apexBtn.setAttribute("aria-pressed", "false");
      Array.prototype.forEach.call(pillars.children, function (c) { c.setAttribute("aria-pressed", "false"); });
    }
    function showApex() {
      clearPressed(); apexBtn.setAttribute("aria-pressed", "true");
      detail.innerHTML = '<h3>Commercial Health</h3><p class="mh-q">' + esc(D.apex.line) + '</p><p>' +
        esc(D.apex.meaning) + '</p><p class="mh-hint">Select a pillar above to see the on-brand material that serves it.</p>' +
        '<a class="btn-cil btn-cil-primary mh-cta" href="' + D.apex.link + '">' + esc(D.apex.link_text) + '</a>';
    }
    function showPillar(id, btn) {
      clearPressed(); if (btn) btn.setAttribute("aria-pressed", "true");
      var p = D.pillars.filter(function (x) { return x.id === id; })[0];
      var html = '<h3>' + esc(p.name) + ' <span style="color:var(--sig-muted);font-weight:400">' + esc(p.sub) +
        '</span></h3><p class="mh-q">' + esc(p.question) + '</p>';
      var any = false;
      D.toolkit.forEach(function (cat) {
        var items = cat.items.filter(function (it) { return (it.pillars || []).indexOf(id) !== -1; });
        if (!items.length) return;
        any = true;
        html += '<div class="mh-cat"><div class="mh-cat-label">' + esc(cat.label) + '</div><ul>' +
          items.map(function (it) { return "<li>" + esc(it.text) + "</li>"; }).join("") + "</ul></div>";
      });
      if (!any) html += '<p class="mh-hint">No material tagged to this pillar yet.</p>';
      detail.innerHTML = html;
    }

    apexBtn.addEventListener("click", showApex);
    pillars.addEventListener("click", function (e) {
      var b = e.target.closest(".mh-pillar"); if (!b) return;
      showPillar(b.getAttribute("data-pillar"), b);
    });
    showApex();
  }
  if (document.readyState === "loading") document.addEventListener("DOMContentLoaded", init);
  else init();
})();
</script>
