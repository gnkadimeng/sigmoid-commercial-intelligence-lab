import sys, pathlib

    _root = _root.parent

from src.utils import load_sample
from src.evidence import summarise_evidence_by_claim
from IPython.display import HTML

register = load_sample("evidence_register.csv")

n_items = len(register)
n_claims = summary["claim_id"].nunique()
n_proposal = int((register["source_type"] == "proposal").sum())

def metric(label, value, sub=""):
    sub_html = f'<div class="metric-sub">{sub}</div>' if sub else ""
    return (f'<div class="metric"><div class="metric-label">{label}</div>'
            f'<div class="metric-value">{value}</div>{sub_html}</div>')

HTML(
    '<div class="metric-row">'
    + metric("Evidence items", n_items, "in the register")
    + metric("Claims tracked", n_claims, "under active enquiry")
    + metric("Proposal-sourced", f"{n_proposal}", "grade-D items (hypothesis only)")
    + metric("Validated claims", "0", "grade A — none yet, by design")
    + "</div>"
)
