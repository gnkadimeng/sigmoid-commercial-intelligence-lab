(function () {
  function init() {
    var el = document.getElementById("chi-data");
    var form = document.getElementById("chi-form");
    if (!el || !form) return;
    var CHI = JSON.parse(el.textContent);
    var labels = (CHI.answer_scale && CHI.answer_scale.labels) || {};
    CHI.dimensions.forEach(function (d) {
      var h = document.createElement("div"); h.className = "chi-dim-head"; h.textContent = d.name;
      form.appendChild(h);
      d.statements.forEach(function (s, i) {
        var row = document.createElement("div"); row.className = "chi-q";
        var t = document.createElement("div"); t.className = "chi-q-text"; t.textContent = s;
        var scale = document.createElement("div"); scale.className = "chi-scale";
        for (var v = 1; v <= 5; v++) {
          var lab = document.createElement("label"); lab.title = labels[v] || v;
          var input = document.createElement("input");
          input.type = "radio"; input.name = d.id + "_" + i; input.value = v;
          if (v === 3) input.checked = true;
          var num = document.createElement("span"); num.textContent = v;
          lab.appendChild(input); lab.appendChild(num); scale.appendChild(lab);
        }
        row.appendChild(t); row.appendChild(scale); form.appendChild(row);
      });
    });
    function bandFor(score) {
      for (var i = 0; i < CHI.bands.length; i++) {
        var b = CHI.bands[i]; if (score >= b.min && score <= b.max) return b;
      }
      return CHI.bands[CHI.bands.length - 1];
    }
    function compute() {
      var wsum = 0, wtot = 0, dims = [];
      CHI.dimensions.forEach(function (d) {
        var vals = d.statements.map(function (s, i) {
          var sel = form.querySelector('input[name="' + d.id + "_" + i + '"]:checked');
          return ((sel ? +sel.value : 3) - 1) / 4 * 100;
        });
        var score = vals.reduce(function (a, b) { return a + b; }, 0) / vals.length;
        dims.push({ name: d.name, score: score, next_move: d.next_move });
        wsum += d.weight * score; wtot += d.weight;
      });
      var chi = Math.round((wsum / wtot) * 10) / 10, band = bandFor(chi);
      var gaps = dims.slice().sort(function (a, b) { return a.score - b.score; }).slice(0, 2);
      document.getElementById("chi-score").innerHTML = Math.round(chi) + '<span style="font-size:1rem;color:var(--sig-muted)"> / 100</span>';
      document.getElementById("chi-conf").textContent = "Confidence " + (CHI.confidence_self_report || "C") + " · self-reported";
      document.getElementById("chi-band").textContent = "Level " + band.level + " · " + band.name;
      document.getElementById("chi-band-meaning").textContent = band.meaning || "";
      document.getElementById("chi-bars").innerHTML = dims.map(function (d) {
        return '<div class="chi-bar-row"><div class="chi-bar-name">' + d.name + '</div><div class="chi-bar-track"><div class="chi-bar-fill" style="width:' + d.score + '%"></div></div><div class="chi-bar-score">' + Math.round(d.score) + '</div></div>';
      }).join("");
      document.getElementById("chi-gaps").innerHTML = '<strong>Start here — your two weakest dimensions:</strong><ul style="margin:.5rem 0 0">' + gaps.map(function (g) {
        return "<li><strong>" + g.name + "</strong> (" + Math.round(g.score) + " / 100) — " + g.next_move + "</li>";
      }).join("") + "</ul>";
    }
    form.addEventListener("change", compute);
    compute();
  }
  if (document.readyState === "loading") document.addEventListener("DOMContentLoaded", init);
  else init();
})();
</script>

## How the score works

Transparent by design — the engine below is the tested `src/chi.py` computing a worked example, so you
can see exactly how a business's answers become a score.

```{python}
#| label: worked-example
#| echo: false
from src.chi import commercial_health_index

sample = {  # an illustrative business (synthetic — for demonstration only)
    "D1": [4, 3, 2], "D2": [3, 2, 3], "D3": [2, 3, 2], "D4": [2, 2, 1],
    "D5": [3, 3, 2], "D6": [2, 2, 3], "D7": [3, 2, 3],
}
res = commercial_health_index(sample)
rows = "".join(
    f"<tr><td>{d['name']}</td><td style='text-align:right'>{d['score']:.0f}</td></tr>"
    for d in res["dimensions"]
)
HTML(
    f"<p><strong>Worked example (illustrative):</strong> a business answering as above scores "
    f"<strong>CHI {res['chi']:.0f} / 100</strong> — Level {res['band']['level']} · {res['band']['name']} "
    f"(confidence {res['confidence']}). Weakest: "
    + ", ".join(f"{g['name']} ({g['score']:.0f})" for g in res['gaps']) + "."
    + "<table class='table' style='margin-top:.6rem;max-width:420px'><thead><tr><th>Dimension</th>"
      "<th style='text-align:right'>Score</th></tr></thead><tbody>" + rows + "</tbody></table>"
)
```

- **Indicator:** each answer `1..5` → `(answer − 1) / 4 × 100`.
- **Dimension:** the mean of its three indicator scores.
- **Commercial Health Index:** the weighted mean of the seven dimensions (equal weight at v0.1).
- **Confidence:** self-assessment is capped at grade **C**; measured data raises it toward **B/A**.
